## Installation

```bash
npx shadcn@latest add https://vitality-shad.vercel.app/r/toggle.json
```

## Usage

```tsx
import { Toggle } from "@/registry/vitality/ui/toggle"
```

```tsx
<Toggle>Toggle</Toggle>
```
